Docker & Compose

Kalypso ships as a Docker image with three ready-to-use Compose setups. Pick the one that fits your stack.

Example 1: Standalone Server

Run Kalypso as a service. Issue certs from the host or other containers via the API.

# examples/standalone/docker-compose.yml

services:
  kalypso:
    image: kalypso/kalypso:latest
    ports:
      - "8200:8200"
    volumes:
      - kalypso-data:/data

volumes:
  kalypso-data:
# Start it
docker compose up -d

# Check health
curl http://localhost:8200/health

# Issue a cert
curl -X POST http://localhost:8200/certificates \
  -H 'Content-Type: application/json' \
  -d '{"domains": ["myapp.local"]}'

Example 2: nginx with Cert Init

Kalypso starts, a one-shot init container requests a cert, then nginx serves HTTPS with the issued cert.

# examples/docker-compose/docker-compose.yml

services:
  kalypso:
    build: ../..         # or image: kalypso/kalypso:latest
    volumes:
      - kalypso-data:/data
    ports:
      - "8200:8200"
    healthcheck:
      test: ["CMD", "python", "-c", "import urllib.request; urllib.request.urlopen('http://localhost:8200/health')"]
      interval: 5s
      timeout: 3s
      retries: 5

  # One-shot: request cert, save to shared volume, exit
  cert-init:
    image: curlimages/curl:latest
    depends_on:
      kalypso:
        condition: service_healthy
    volumes:
      - certs:/certs
    entrypoint: /bin/sh
    command:
      - -c
      - |
        set -e
        RESPONSE=$$(curl -sf http://kalypso:8200/certificates \
          -H 'Content-Type: application/json' \
          -d '{"domains": ["myapp.local", "*.myapp.local"], "hours": 24}')
        echo "$$RESPONSE" | sed -n 's/.*"certificate":"\([^"]*\)".*/\1/p' \
          | sed 's/\\n/\n/g' > /certs/cert.pem
        echo "$$RESPONSE" | sed -n 's/.*"private_key":"\([^"]*\)".*/\1/p' \
          | sed 's/\\n/\n/g' > /certs/key.pem

  web:
    image: nginx:alpine
    depends_on:
      cert-init:
        condition: service_completed_successfully
    ports:
      - "443:443"
      - "80:80"
    volumes:
      - certs:/etc/nginx/certs:ro
      - ./nginx.conf:/etc/nginx/conf.d/default.conf:ro

volumes:
  kalypso-data:
  certs:

Example 3: Auto-Refresh Sidecar

Like Example 2, but the cert-refresh sidecar keeps running and renews the cert every 12 hours. Certs never expire during long dev sessions.

# examples/docker-compose-auto-refresh/docker-compose.yml

services:
  kalypso:
    build: ../../..
    volumes:
      - kalypso-data:/data
    ports:
      - "8200:8200"
    healthcheck:
      test: ["CMD", "python", "-c", "import urllib.request; ..."]
      interval: 5s

  # Long-running sidecar: fetch cert now, then every 12 hours
  cert-refresh:
    image: curlimages/curl:latest
    depends_on:
      kalypso:
        condition: service_healthy
    volumes:
      - certs:/certs
    entrypoint: /bin/sh
    command:
      - -c
      - |
        fetch_cert() {
          RESPONSE=$$(curl -sf http://kalypso:8200/certificates \
            -H 'Content-Type: application/json' \
            -d '{"domains": ["myapp.local", "*.myapp.local"], "hours": 24}')
          # ... parse and save cert.pem / key.pem ...
        }
        fetch_cert                 # initial fetch
        while true; do
          sleep 43200              # 12 hours
          fetch_cert
        done

  web:
    image: nginx:alpine
    depends_on:
      cert-refresh:
        condition: service_started
    volumes:
      - certs:/etc/nginx/certs:ro
      - ./nginx.conf:/etc/nginx/conf.d/default.conf:ro

volumes:
  kalypso-data:
  certs:

Trusting the CA from Docker

After starting the compose stack, grab the CA cert and trust it on your host machine:

# Download the CA cert from the running Kalypso server
curl -s http://localhost:8200/ca.pem | jq -r .certificate > kalypso-ca.pem

# macOS
sudo security add-trusted-cert -d -r trustRoot \
  -k /Library/Keychains/System.keychain kalypso-ca.pem

# Ubuntu / Debian
sudo cp kalypso-ca.pem /usr/local/share/ca-certificates/kalypso.crt
sudo update-ca-certificates

# Don't forget /etc/hosts
echo "127.0.0.1 myapp.local" | sudo tee -a /etc/hosts

Trusting inside other containers

If container A needs to call container B over HTTPS with a Kalypso cert, mount the CA cert:

services:
  my-service:
    environment:
      - REQUESTS_CA_BUNDLE=/etc/ssl/certs/kalypso-ca.pem  # Python requests
      - NODE_EXTRA_CA_CERTS=/etc/ssl/certs/kalypso-ca.pem  # Node.js
      - SSL_CERT_FILE=/etc/ssl/certs/kalypso-ca.pem       # Ruby, Go
    volumes:
      - kalypso-ca:/etc/ssl/certs/kalypso-ca.pem:ro
